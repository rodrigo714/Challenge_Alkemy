public byte[] ConvertToBytes(HttpPostedFileBase image)
        {
            try
            {
                byte[] imageBytes;
                BinaryReader reader = new BinaryReader(image.InputStream);
                imageBytes = reader.ReadBytes((int)image.ContentLength);
                return imageBytes;
            }
            catch (Exception)
            {
                throw;
            }
            
        }
